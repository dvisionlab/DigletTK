/**
* Example of adding a surface in a VR scene
*/

const element = document.getElementById("viewer-2");

const vr = new diglettk.VRView(element);

fetch("./demo/die.stl").then(response => {
  response.arrayBuffer().then(data => {
    vr.addSurface({ buffer: data, color: [0, 0, 1], label: "dice" });
  });
});
