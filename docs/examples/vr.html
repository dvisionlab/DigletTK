const element = document.getElementById('viewer-2')

diglettk.loadSerieWithLarvitar(serie => {
    // build vtk volume with larvitar
    const image = diglettk.buildVtkVolume(serie);
    // run vr
    let vr = new diglettk.VRView(element);
    // set an image
    vr.setImage(image);
})
